const chalk = require('chalk');
const def = chalk.bold.white;
const info = chalk.bold.yellow;
const progress = chalk.bold.green;
const warning = chalk.bold.keyword('orange');
const error = chalk.inverse.red;
/**
 * Custom logging function for lesslms-cli tool.
 * @function
 * @param {string} string String to log out to console.
 * @param {string} level Type of log: 'info' | 'progress' | 'warning' | 'error'. 
 */
function _log(string, level) {
    const supportsColor = require('supports-color');
    _chalk = def;
    switch (level) {
        case 'info':
            _chalk = info;
            break;

        case 'progress':
            _chalk = progress;
            break;

        case 'warning':
            _chalk = warning;
            break;

        case 'error':
            _chalk = error;
            break;

        default:
            break;

    }
    console.log(_chalk(string));
}

module.exports = _log;
